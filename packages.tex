\newcommand{\boxnode}[2]
{
    {
    %------------ Styling ----------------
    
    %------------ Content ----------------
    \begin{tikzpicture}
    \node [mybox] (box){%
        \begin{minipage}{0.3\textwidth}
                #2
        \end{minipage}
    };
    %------------  Header ---------------------
    \node[fancytitle, right=10pt] at (box.north west) {#1};
    \end{tikzpicture}
    }
}
